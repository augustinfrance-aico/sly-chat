Python — AI-Powered Flight Scoring (Excerpt)

class FlightScorer:
    # Score each flight 0-100 based on multiple criteria

    def score_flight(self, flight: dict) -> int:
        score = 100

        # Price vs average (40% weight)
        avg_price = self.get_historical_avg(flight['route'])
        price_ratio = flight['price'] / avg_price
        if price_ratio < 0.8:
            score += 15   # 20%+ below average = bonus
        elif price_ratio > 1.2:
            score -= 20   # 20%+ above average = penalty

        # Stops penalty (25% weight)
        score -= flight['stops'] * 12

        # Duration penalty (20% weight)
        if flight['duration_hrs'] > 24:
            score -= (flight['duration_hrs'] - 24) * 3

        # Preferred airline bonus (15% weight)
        if flight['airline'] in self.preferred:
            score += 8

        return max(0, min(100, score))

    async def daily_scan(self):
        # Scrape all sources in parallel
        results = await asyncio.gather(
            self.scrape_google_flights('DFW', 'SYD', '2026-12'),
            self.scrape_skyscanner('DFW', 'SYD', '2026-12'),
            self.scrape_kayak('DFW', 'SYD', '2026-12'),
        )

        # Score, rank, and alert if deal found
        scored = [self.score_flight(f) for f in results]
        deals = [f for f in scored if f['score'] >= 80]

        if deals:
            await self.send_alert(deals)  # Email + Telegram
